--ROOT FOLDER

SELECT 
	F.ID, 
	F.NAME,
	F.CREATED,
	F.DISABLED, 
	F.MODIFIED,
	F.VISIBLE  
FROM 
	FOLDER F 
WHERE
	F.NAME = 'root'


--CHILDREN

SELECT 
	F.ID, 
	F.NAME,
	F.CREATED,
	F.DISABLED, 
	F.MODIFIED,
	F.VISIBLE 
FROM 
	FOLDER F 
	INNER JOIN FOLDER_FOLDERS FF ON
		F.ID = FF.FOLDERS_ID
WHERE
	FF.FOLDER_ID = 2
